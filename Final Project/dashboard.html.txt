<script>
const client=mqtt.connect('wss://broker.hivemq.com:8000/mqtt');
const chart=new Chart(document.getElementById('chart'),{type:'line',data:{labels:[],datasets:[{label:'Temp',borderColor:'red',data:[]},{label:'Hum',borderColor:'blue',data:[]}]}});
let readings=[];
client.on('connect',()=>{document.getElementById('status').innerHTML='✅ LIVE';document.getElementById('status').style.color='green';client.subscribe('example/topic');});
client.on('message',(t,p)=>{const d=JSON.parse(p);const time=new Date().toLocaleTimeString();readings.unshift({time,temp:d.temp,hum:d.hum});if(readings.length>20)readings.pop();document.getElementById('data').innerHTML='<tr><th>Time</th><th>Temp</th><th>Hum</th></tr>'+readings.map(r=>`<tr><td>${r.time}</td><td>${r.temp}</td><td>${r.hum}</td></tr>`).join('');chart.data.labels=readings.map(r=>r.time);chart.data.datasets[0].data=readings.map(r=>r.temp);chart.data.datasets[1].data=readings.map(r=>r.hum);chart.update();});
</script></body></html>
